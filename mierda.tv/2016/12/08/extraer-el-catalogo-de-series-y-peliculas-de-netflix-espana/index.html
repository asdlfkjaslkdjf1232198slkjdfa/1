Extraer el catálogo de series y películas de Netflix España

8 diciembre, 2016 En # CODE

Sin necesidad de cuenta en Netflix ni tener que visitar la web de Netflix es posible extraer la información del catalogo de Netflix España y darse cuenta de que es una mierda realmente comparado con por ejemplo filmin. Básicamente en Netflix se encuentra basura mayormente.

Con este sencillo script en bash podemos extraer el contenido de quienes lo extraen previamente.

netflix.tar.gz

Dejo el script pero mejor bájate el tar.gz y ejecútalo.

#!/bin/bash
# script en bash para obtener listado de peliculas y series en netflix españa
# Una prueba de concepto para obtener el listado.
# Dependencias: apt-get install jshon
# Crear un directorio llamado imgs para que por defecto entren allí las imagenes
# fanta <fantanaranja@openmailbox.org>


# Variables iniciales (no tocar si no sabes que demonios haces)

maxdraw=30 # número de peticiones maximas para obtener datos. 30x100(lenght) = 3000 pelis/series/docus
start=0
length=100 # incrementará $start en este valor
dirimgs="imgs" # directorio en el que se guardarán las imgs
output="netflix.csv" # la salida con el contenido separado por ;


for a in `seq 0 $maxdraw`;
do
    wget -q "http://es.allflicks.net/wp-content/themes/responsive/processing/processing.php?draw=$a&columns%5B0%5D%5Bdata%5D=box_art&columns%5B0%5D%5Bname%5D=&columns%5B0%5D%5Bsearchable%5D=true&columns%5B0%5D%5Borderable%5D=false&columns%5B0%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B0%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B1%5D%5Bdata%5D=title&columns%5B1%5D%5Bname%5D=&columns%5B1%5D%5Bsearchable%5D=true&columns%5B1%5D%5Borderable%5D=true&columns%5B1%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B1%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B2%5D%5Bdata%5D=year&columns%5B2%5D%5Bname%5D=&columns%5B2%5D%5Bsearchable%5D=true&columns%5B2%5D%5Borderable%5D=true&columns%5B2%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B2%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B3%5D%5Bdata%5D=genre&columns%5B3%5D%5Bname%5D=&columns%5B3%5D%5Bsearchable%5D=true&columns%5B3%5D%5Borderable%5D=true&columns%5B3%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B3%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B4%5D%5Bdata%5D=rating&columns%5B4%5D%5Bname%5D=&columns%5B4%5D%5Bsearchable%5D=true&columns%5B4%5D%5Borderable%5D=true&columns%5B4%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B4%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B5%5D%5Bdata%5D=available&columns%5B5%5D%5Bname%5D=&columns%5B5%5D%5Bsearchable%5D=true&columns%5B5%5D%5Borderable%5D=true&columns%5B5%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B5%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B6%5D%5Bdata%5D=director&columns%5B6%5D%5Bname%5D=&columns%5B6%5D%5Bsearchable%5D=true&columns%5B6%5D%5Borderable%5D=true&columns%5B6%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B6%5D%5Bsearch%5D%5Bregex%5D=false&columns%5B7%5D%5Bdata%5D=cast&columns%5B7%5D%5Bname%5D=&columns%5B7%5D%5Bsearchable%5D=true&columns%5B7%5D%5Borderable%5D=true&columns%5B7%5D%5Bsearch%5D%5Bvalue%5D=&columns%5B7%5D%5Bsearch%5D%5Bregex%5D=false&order%5B0%5D%5Bcolumn%5D=5&order%5B0%5D%5Bdir%5D=desc&start=$start&length=$length&search%5Bvalue%5D=&search%5Bregex%5D=false&movies=true&shows=true&documentaries=true&rating=netflix&min=1900&max=2016&_=1480444843093" -O temporal.json
    npelis=$(cat temporal.json | jshon -e data -l) # número de pelis/series a procesar
    for b in `seq 0 $npelis`;
    do
        # Obtenemos el id de la pelicula/serie, el titulo y el año
        id=$(cat temporal.json | jshon -e data -e $b | grep "\"id\":" | cut -d "\"" -f 4)
        titulo=$(cat temporal.json | jshon -e data -e $b | grep "\"title\":" | cut -d "\"" -f 4)
        ano=$(cat temporal.json | jshon -e data -e $b | grep "\"year\":" | cut -d "\"" -f 4)
        # Obtenemos el genero de la pelicula/serie, el/la director/a y el reparto
        genero=$(cat temporal.json | jshon -e data -e $b | grep "\"genre\":" | rev | cut -d "<" -f 2 | cut -d ">" -f 1 | rev)
        director=$(cat temporal.json | jshon -e data -e $b | grep "\"director\":" | cut -d "\"" -f 4)
        reparto=$(cat temporal.json | jshon -e data -e $b | grep "\"cast\":" | cut -d "\"" -f 4)
        # Obtenemos la imagen de la pelicula/serie/documental/... y la descargamos
        imgurl=$(cat temporal.json | jshon -e data -e $b | grep "\"box_art\":" | cut -d "\"" -f 5 | sed 's/\\//g')
        nombreimagen=$(cat temporal.json | jshon -e data -e $b | grep "\"box_art\":" | cut -d "\"" -f 5 | sed 's/\\//g' | rev | cut -d "/" -f 1 | rev)
        wget -q "$imgurl" -O $dirimgs/$nombreimagen
        # Obtenemos la descripción de la pelicula/serie/documental
        descripcion=$(wget -q "http://es.allflicks.net/peliculas/$id/" -O -| grep "class=\"borderless\"" | tail -1 | cut -d ">" -f 2 | cut -d "<" -f 1)
        echo "$id;$titulo;$ano;$genero;$director;$reparto;$descripcion;$nombreimagen"
        # Vamos grabando la salida línea a línea
        echo "$id;$titulo;$ano;$genero;$director;$reparto;$descripcion;$nombreimagen" >> $output
        sleep 1
    done

    let start=$start+$length
done
